// ---------- ENHANCED MATCHING VISUALIZATION ---------- const canvas = document.getElementById('matchingCanvas'); const container = document.getElementById('matchingVizContainer'); const tooltip = document.getElementById('matchTooltip'); let ctx = canvas.getContext('2d'); let width = 0, height = 0; let hoveredNode = null; // { type, index, x, y, label } let animationId = null; let pulses = []; // ---- Refined brand & creator nodes (aligned with Brioo's real focus) ---- const brandNodes = [ { label: '🌿 Ethical Beauty', type: 'brand', x: 0.18, y: 0.25, matchScore: 94 }, { label: '📈 Fintech', type: 'brand', x: 0.18, y: 0.45, matchScore: 88 }, { label: '🏋️‍♂️ Modest Fitness', type: 'brand', x: 0.18, y: 0.65, matchScore: 91 }, { label: '📚 Online Education', type: 'brand', x: 0.18, y: 0.85, matchScore: 79 } ]; const creatorNodes = [ { label: '✨ Micro‑Influencer', type: 'creator', x: 0.82, y: 0.20, matchScore: 96 }, { label: '💼 Finance Educator', type: 'creator', x: 0.82, y: 0.40, matchScore: 90 }, { label: '🧘 Wellness Coach', type: 'creator', x: 0.82, y: 0.60, matchScore: 87 }, { label: '🕌 Faith Lifestyle', type: 'creator', x: 0.82, y: 0.80, matchScore: 93 } ]; // Combine for mouse hit testing const allNodes = [...brandNodes, ...creatorNodes]; function resizeCanvas() { const rect = container.getBoundingClientRect(); canvas.width = rect.width; canvas.height = rect.height; width = rect.width; height = rect.height; draw(); } window.addEventListener('resize', resizeCanvas); resizeCanvas(); // Helper: get absolute canvas coordinates for a node function getNodePos(node) { return { x: node.x * width, y: node.y * height }; } // Draw the whole scene function draw() { if (!width || !height) return; ctx.clearRect(0, 0, width, height); // ---- Background (matches dark theme) ---- ctx.fillStyle = '#0a0a0f'; ctx.fillRect(0, 0, width, height); // ---- Draw all connection lines (static) ---- ctx.beginPath(); ctx.lineWidth = 1.2; for (let b of brandNodes) { for (let c of creatorNodes) { const p1 = getNodePos(b); const p2 = getNodePos(c); ctx.beginPath(); ctx.moveTo(p1.x, p1.y); ctx.lineTo(p2.x, p2.y); ctx.strokeStyle = 'rgba(200,164,107,0.2)'; ctx.stroke(); } } // ---- Draw pulsing active lines (dynamic matches) ---- for (let p of pulses) { const b = brandNodes[p.brandIdx]; const c = creatorNodes[p.creatorIdx]; const p1 = getNodePos(b); const p2 = getNodePos(c); ctx.beginPath(); ctx.moveTo(p1.x, p1.y); ctx.lineTo(p2.x, p2.y); const intensity = 0.4 + p.progress * 0.6; ctx.strokeStyle = `rgba(200,164,107,${intensity})`; ctx.lineWidth = 3 + p.progress * 2; ctx.stroke(); p.progress += 0.025; } pulses = pulses.filter(p => p.progress < 1); // ---- Draw the Brioo logo / central hub ---- const centerX = width / 2; const centerY = height / 2; ctx.save(); ctx.shadowBlur = 15; ctx.shadowColor = 'rgba(200,164,107,0.6)'; ctx.beginPath(); ctx.arc(centerX, centerY, 28, 0, 2 * Math.PI); ctx.fillStyle = '#13131a'; ctx.fill(); ctx.strokeStyle = '#c8a46b'; ctx.lineWidth = 2; ctx.stroke(); ctx.font = `bold ${Math.min(18, width/40)}px 'Playfair Display'`; ctx.fillStyle = '#c8a46b'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText('B', centerX, centerY); ctx.restore(); // ---- Draw all nodes (brands left, creators right) with hover glow ---- for (let node of allNodes) { const pos = getNodePos(node); const isHovered = (hoveredNode && hoveredNode.label === node.label); ctx.beginPath(); ctx.arc(pos.x, pos.y, 14, 0, 2 * Math.PI); ctx.fillStyle = isHovered ? '#1e1e2a' : '#13131a'; ctx.fill(); ctx.strokeStyle = isHovered ? '#e0c49a' : '#c8a46b'; ctx.lineWidth = isHovered ? 2.5 : 1.5; ctx.stroke(); // Inner glow for hover if (isHovered) { ctx.beginPath(); ctx.arc(pos.x, pos.y, 16, 0, 2 * Math.PI); ctx.fillStyle = 'rgba(200,164,107,0.15)'; ctx.fill(); } // Node label ctx.font = `500 ${Math.min(11, width/70)}px 'DM Sans'`; ctx.fillStyle = isHovered ? '#e0c49a' : '#ffffffcc'; ctx.textAlign = 'center'; ctx.fillText(node.label, pos.x, pos.y - 18); } // ---- Emit new random pulse every 1.8 seconds ---- if (Math.random() < 0.02) { // approximate const brandIdx = Math.floor(Math.random() * brandNodes.length); const creatorIdx = Math.floor(Math.random() * creatorNodes.length); pulses.push({ brandIdx, creatorIdx, progress: 0 }); } animationId = requestAnimationFrame(draw); } draw(); // ---- Mouse hover detection and tooltip ---- function handleMouseMove(e) { const rect = canvas.getBoundingClientRect(); const mouseX = e.clientX - rect.left; const mouseY = e.clientY - rect.top; let found = null; for (let node of allNodes) { const pos = getNodePos(node); const dx = mouseX - pos.x; const dy = mouseY - pos.y; const dist = Math.sqrt(dx*dx + dy*dy); if (dist < 20) { found = node; break; } } if (found) { hoveredNode = found; const pos = getNodePos(found); // Show match score tooltip const matchInfo = found.type === 'brand' ? `🤝 Matches ${creatorNodes.length} creators · ${found.matchScore}% avg fit` : `⭐ Brand demand · ${found.matchScore}% potential`; tooltip.style.display = 'block'; tooltip.style.left = (pos.x + 20) + 'px'; tooltip.style.top = (pos.y - 30) + 'px'; tooltip.innerHTML = matchInfo; } else { hoveredNode = null; tooltip.style.display = 'none'; } } canvas.addEventListener('mousemove', handleMouseMove); canvas.addEventListener('mouseleave', () => { hoveredNode = null; tooltip.style.display = 'none'; }); // ---- Also update on window resize to reposition nodes ---- window.addEventListener('resize', () => { resizeCanvas(); // Force tooltip hide tooltip.style.display = 'none'; hoveredNode = null; });
